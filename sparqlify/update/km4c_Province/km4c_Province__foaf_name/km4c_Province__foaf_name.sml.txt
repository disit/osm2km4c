Prefix foaf:<http://xmlns.com/foaf/0.1/>

Create View ProvinceName As

Construct {
Graph ?graph_uri {
?s foaf:name ?name .
}}

With
?graph_uri = uri(?graph_uri)
?s = uri(concat("http://www.disit.org/km4city/resource/", ?id))
?name = plainLiteral(?p_name)

From [[
select * from ProvinceName
]]
